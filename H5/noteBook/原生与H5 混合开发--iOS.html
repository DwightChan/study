关闭当前页面

  • eg:
/**
 *  关闭当前界面
 *
 *  @param 入参 params 可不传
 *  @return 返回值 无
 *  @example 实例 QF_Nativ.ntvCloseWebview();
 */
QF_Native.ntvCloseWebview = function(params) {
    qfApi("closeWebview",params);
};

获取用户登录信息

  • eg:
/**
 *  获取用户登录信息
 *
 *  @param 入参 params 可不传
 *  @return 返回值 无
 *  @example 实例 QF_Native.ntvGetLoginInfo();
 *  {
 *      isLogin: "true", // true登录, false 未登录
 *      account: "test01", // 用户名
 *      userNumber: "15900000001", // 用户手机号
 *      mac: "xxxxxxxxx",   // 物理地址
 *      token: "ios||1568124277||573c161fbec4ec008081172b29167965cade550a", // token
 *      userMd5: "",    // 用户名 MD5值
 *      userDes3: "", // 3重向量加密
 *  }
 */
QF_Native.ntvGetLoginInfo = function(params) {
    qfApi("getLoginInfo",params);
};

调用原生去登录

  • eg:
/**
 *  去登录
 *
 *  @param 入参 params 可不传
 *  @return 返回值 无
 *  @example 实例 QF_Native.ntvToLogin("");
 *  需要原生登录完成之后 回传用户信息给h5 时 必传参数 : json格式数据
 *  {
 *       "isGetLoginInfo": true, // 有 参数 "isGetLoginInfo": true 并且为true时, 要回传 用户信息给H5
 *       "hudStr": "文本提示内容", // 文本提示内容, 没有可以不传
 *       "isRefresh": true, // true: reload web页面, false: 无动作
 *  }
 */
QF_Native.ntvToLogin = function(params) {
    qfApi("toLogin",params);
};

账号已经下线(被挤下线/ 踢下线)

  • eg: QF_Native.ntvAlreadyLogout("");
/**
 *  在h5 页面下线了
 *
 *  @param 入参 params 可不传
 *  @return 返回值 无
 *  @example 实例 QF_Native.ntvAlreadyLogout("");
 需要原生登录完成之后 回传用户信息给h5 时 必传参数 : json格式数据
 *  {
 *       "hudStr": "文本提示内容", // 文本提示内容, 没有可以不传
 *  }
 */
QF_Native.ntvAlreadyLogout = function(params) {
    //qfApi("toLogin",params);
    window.webkit.messageHandlers.alreadyLogout.postMessage(JSON.stringify(params));
};

跳转优惠记录

  • eg: QF_Native.ntvToPromotionRecord("");
/**
 *  跳转优惠记录 toPromotionRecord
 *
 *  @param 入参 params 必传 json 对象
 *  @return 返回值 无
 *  @example 实例 QF_Native.ntvToPromotionRecord("");
 */
QF_Native.ntvToPromotionRecord = function(params) {
    // JSON.stringify(params) 对象转为 string
    window.webkit.messageHandlers.toPromotionRecord.postMessage(JSON.stringify(params));
};
